	var i;

	values = [

	];

		t.throws( badValue( values[ i ] ), Error, 'throws an error when provided '+values[ i ] );
	}
	t.end();

	function badValue( value ) {
		return function badValue() {
			{{ALIAS}}( value );
		}
	}
});

tape( 'the function {{DESC}} (primitive)', function test( t ) {
	var expected;
	var actual;
	var values;
	var v;
	var i;

	values = [ {{VALUES_LEN_4}} ];

	for ( i = 0; i < values.length; i++ ) {
		v = values[ i ];
		actual = {{ALIAS}}( v );
		expected = ref( v );

	}
	t.end();
});

tape( 'the function {{DESC}} (nested array)', function test( t ) {
	var expected;
	var actual;
	var values;
	var arr;
	var tmp;
	var v;
	var i;
	var j;

	values = [ {{VALUES_LEN_4}} ];

	arr = [];
	expected = [];
	for ( i = 0; i < 10; i++ ) {
		tmp = [];
		for ( j = 0; j < 10; j++ ) {
			v = values[ j%values.length ];
			tmp.push( v );
			expected.push( ref( v ) );
		}
		arr.push( tmp );
	}
	actual = {{ALIAS}}( arr );

	t.strictEqual( isArray( actual ), true, 'returns expected value' );
	t.strictEqual( isArray( actual[ 0 ] ), true, 'returns expected value' );
	t.deepEqual( flatten2d( actual ), expected, 'returns expected value' );
	t.end();
});
